def msg():
    val1 = int(input("Enter first value: "))
    val2 = int(input("Enter second value: "))
    sum=val1+val2
    sub=val1-val2
    div=val1*val2
    return sum, sub, div

#positional argument
def personalInfo(fname, lname):
    print("first name: ", fname)
    print("last name: ", lname)
personalInfo("Prashant", "Jha")






#keyword argument
def personalInfo(fname, lname):
    print("first name: ", fname)
    print("last name: ", lname)
    
fname = "Prashant"
lname = "Jha"
personalInfo(fname, lname)








#kdefault argument
def cityName(city="Nagpur"):
    print("city name: ", city)

cityName("Mumbai")
cityName("Delhi")
cityName()







#varible length argument
def cityName(*city):
    print("city name:", city)

cityName("Mumbai", "Nagpur", "Pune", "Nashik")
